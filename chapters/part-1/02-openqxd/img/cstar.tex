                    }
                }{
                    \ifthenelse{\y=0} {
                        \filldraw[color=cgreen,fill=cgreen](n) circle (0.2); % physical
                    } {
                        \filldraw[color=black,fill=black](n) circle (0.2); % physical
                    }
